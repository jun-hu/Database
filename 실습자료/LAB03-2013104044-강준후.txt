SELECT employee_id, last_name, salary, salary*1.15 as "New Salary"
FROM employees;

SELECT last_name as name, length(last_name) as name_length
FROM employees
where SUBSTR(last_name,1,1) in ('A','J','M')
order by last_name ASC;
